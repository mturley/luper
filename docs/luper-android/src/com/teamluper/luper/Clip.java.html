  public void getDuration() throws IOException
  {
    MediaPlayer mp = new MediaPlayer();
    FileInputStream fs;
    FileDescriptor fd;
    fs = new FileInputStream(name);
    fd = fs.getFD();
    mp.setDataSource(fd);
    mp.prepare();
    int length = mp.getDuration();
    mp.release();
    duration = length;
  }
